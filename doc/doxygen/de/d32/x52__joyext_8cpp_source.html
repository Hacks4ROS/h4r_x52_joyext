90 void X52_JoyExt::cb_mfd_text(const x52_joyext::x52_mfdConstPtr &msg)
91 {
92  int curPos=msg->pos+16*msg->line;
93 
94  if(msg->clearDisplay)
95  {
96  mfd_content[0]=" ";
97  mfd_content[1]=" ";
98  mfd_content[2]=" ";
99  memset(updateMFD,true,3);
100  }
101 
102  for (std::string::const_iterator it=msg->data.begin(); it != msg->data.end(); ++it)
103  {
104  char curChr=*it;
105 
106  if(curPos<3*16)
107  {
108  uint8_t curLine=curPos/16;
109  uint8_t posInLine=curPos-curLine*16;
110  if(curChr=='\n')
111  {
112  //Fill the rest of the line with spaces
113  for (curPos = 0; curPos < (curLine+1)*16; ++curPos)
114  {
115  posInLine=curPos-curLine*16;
116  mfd_content[curLine][posInLine]=curChr;
117  }
118  continue;
119  }
120  else
121  {
122  mfd_content[curLine][posInLine]=curChr;
123  updateMFD[curLine]=true;
124  curPos++;
125  }
126  }
127  else
128  {
129  ROS_WARN("MFD text exceeds display bounds!");
130  break;
131  }
132  }
133  ROS_DEBUG("Line 0: %s", mfd_content[0].c_str());
134  ROS_DEBUG("Line 1: %s", mfd_content[1].c_str());
135  ROS_DEBUG("Line 2: %s", mfd_content[2].c_str());
136 }

186 void X52_JoyExt::send_to_joystick()
187 {
188  /*Get Joystick */
189  struct x52 *hdl=0;
190  while(hdl==0 && ros::ok())
191  {
192  ROS_INFO("Trying to find joystick...");
193  hdl=x52_init();
194  loop_rate->sleep();
195  }
196  ROS_INFO("Joystick found...");
197  x52_debug(hdl, 1);
198 
199 
200  x52_setled(hdl,0,0);
201 
202  while(ros::ok())
203  {
204  //Update Joystick LED
205  for (int lednum = 0; lednum < 10; ++lednum)
206  {
207  if(updateLED[lednum])
208  {
209  ROS_DEBUG("Setting LED: %i",lednum);
210  if(lednum != 0)
211  {
212  int led=(lednum)*2;
213  x52_setled(hdl,led,LED[led-1]);
214  x52_setled(hdl,led+1,LED[led]);
215  }
216  else
217  {
218  x52_setled(hdl,1,LED[0]);
219  }
220  updateLED[lednum]=false;
221  usleep(50);
222  }
223  }
224 
225  //Update Brightness MFD
226  if(updateBrightnessMFD)
227  {
228  x52_setbri(hdl,1,brightnessMFD);
229  updateBrightnessMFD=false;
230  usleep(50);
231  }
232 
233  //Update Brightness LED
234 
235  if(updateBrightnessLED)
236  {
237  x52_setbri(hdl,0,brightnessLED);
238  updateBrightnessLED=false;
239  usleep(50);
240  }
241 
242  //Update MFD Text
243  for (int line = 0; line < 3; ++line)
244  {
245  if(updateMFD[line])
246  {
247  x52_settext(hdl,line,(char*)mfd_content[line].c_str(),mfd_content[line].length());
248  updateMFD[line]=false;
249  }
250  usleep(50);
251  }
252 
253  //Update Date
254  if(updateDate)
255  {
256  x52_setdate(hdl,Date[0],Date[1],Date[2]);
257  updateDate=false;
258  usleep(50);
259  }
260 
261  //Update Time
262  if(updateTime)
263  {
264  x52_settime(hdl,Time_24,Time[0], Time[1]);
265  updateTime=false;
266  usleep(50);
267  }
268 
269  //Offset Time
270  for (int o = 0; o < 2; ++ o)
271  {
272  if(updateOffset[o])
273  {
274  x52_setoffs(hdl,o,Offset_24[o],Offset_Inv[o],Offset[o]);
275  updateOffset[o]=false;
276  }
277  usleep(50);
278  }
279 
280  ros::spinOnce();
281  loop_rate->sleep();
282  }
283 }
